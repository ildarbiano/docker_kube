=========== kubectl nodes/pods ====================================================================
kubectl get nodes				# после запуска, Должна быть нода. Использует системный kubectl (который вы установили). Когда конфигурация настроена правильно. kubectl - "обычный" способ, который требует правильной настройки.
minikube kubectl -- get nodes -o wide		# после запуска, Должна быть нода. Использует kubectl ВНУТРИ minikube контейнера/VM. Версия kubectl совместима с версией Kubernetes в 	minikube. Гарантированно работает с текущим кластером minikube. Не зависит от вашей системной установки kubectl. minikube kubectl -- - это "безопасный" способ. Полезно. 	Используйте minikube kubectl -- когда:
minikube kubectl -- get nodes			# Когда ваш системный kubectl не настроен
kubectl get nodes				# Проверяем кластер
kubectl cluster-info				# Проверяем кластер
kubectl get pods				# 
kubectl get pods -A				# Для повседневной работы (удобнее)
kubectl get pods -n homework -o wide		# Проверим что поды действительно работают. Посмотрим IP подов. Должны увидеть те же IP, что в Endpoints.
kubectl logs deployment/web			# Посмотрите логи пода
	2025/11/03 13:36:47 Server listening on port 8080
kubectl logs <pod-name>				# Проверить логи приложения. # Посмотреть логи пода (замените на имя вашего пода)
kubectl logs <pod-name> -n homework		# Если под в другом namespace
kubectl logs -f <pod-name>			# Посмотреть логи в реальном времени

context	namespace 				- это "текущая рабочая папка" в Kubernetes. Как в командной строке вы можете быть в разных папках, так и в kubectl вы можете работать в разных namespace.
context	namespace 				- это как "текущая папка" в Kubernetes. Когда вы устанавливаете namespace, все команды kubectl выполняются в этом namespace, если явно не указано иное.
kubectl config view				# Посмотреть полную информацию о контексте
kubectl config current-context			# Проверьте текущий контекст
kubectl config get-contexts			# Посмотрите все доступные контексты
	Зачем это нужно?
	Безопасность - не видите чужие ресурсы
	Удобство - не нужно постоянно указывать -n homework
	Организация - логическое разделение проектов
kubectl config set-context --current --namespace=homework		# Установить homework как namespace по умолчанию
kubectl config view --minify --output 'jsonpath={..namespace}'		# Проверить текущий namespace
kubectl delete namespace homework		# Удаляем все ресурсы в namespace homework

----------- Ingress ---------------------------------------
Ingress 					- это профессиональное решение для доступа к сервисам 
						- это правильное решение для доступа к сервисам по доменным именам. В Minikube нужно включить Ingress контроллер.
Установка и настройка Ingress в Minikube. Включите Ingress контроллер в Minikube
powershell
Addon (аддон) 					— это дополнительный компонент, который улучшает или расширяет функциональность существующей программы, игры или системы. Например, в играх аддоны добавляют новый контент, а в браузере — новые функции или инструменты. 
minikube addons disable ingress			# 
minikube addons enable ingress			# Включить ingress addon
minikube addons list | findstr ingress		# Проверить статус. Это просто статус настроек Minikube, а не реальное состояние кластера.
kubectl get ingress -n homework		
kubectl get pods -n ingress-nginx		# Показывает реальные работающие поды в кластере.
kubectl delete ingress example-ingress -n homework
kubectl apply -f .\ingress.yaml			# Применить ingress ресурс
kubectl apply -f ingress.yaml
kubectl get ingress -n homework			# Проверить
kubectl get ingress -A				# Проверим текущую конфигурацию ingress
kubectl get ingress example-ingress -n homework -o yaml
kubectl delete ingress web-ingress -n homework	# Удалить ingress из правильного namespace
